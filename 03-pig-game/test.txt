"use strict";
let scorePlayerOne = document.querySelector("#score--0");
let scorePlayerTwo = document.querySelector("#score--1");
let currenScoreOne = document.getElementById("current--0");
let currenScoreTwo = document.getElementById("current--1");
const playerOne = document.querySelector(".player--0");
const playerTwo = document.querySelector(".player--1");
const diceImage = document.querySelector(".dice");
const btnRoll = document.querySelector(".btn--roll");
const btnNew = document.querySelector(".btn--new");
const btnHold = document.querySelector(".btn--hold");
diceImage.classList.add("hidden");

let activePlayer = 0;

let currentScore = 0;
function handleRoll() {
  // Generating a random dice roll
  let dice = Math.trunc(Math.random() * 6) + 1;
  diceImage.classList.remove("hidden");
  diceImage.src = `dice-${dice}.png`;
  if (dice !== 1) {
    if (activePlayer === 0) {
      currentScore += dice;
      currenScoreOne.textContent = currentScore;
    } else {
      currentScore += dice;
      currenScoreTwo.textContent = currentScore;
    }
  } else {
    if (playerTwo.classList.contains("player--active")) {
      playerTwo.classList.remove("player--active");
      playerOne.classList.add("player--active");
      activePlayer = 0;
      currentScore = 0;
      currenScoreTwo.textContent = 0;
    } else {
      playerTwo.classList.add("player--active");
      playerOne.classList.remove("player--active");
      activePlayer = 1;
      currentScore = 0;
      currenScoreOne.textContent = 0;
    }
  }
  console.log(dice);
}

function newGame() {
  scorePlayerOne.textContent = 0;
  scorePlayerTwo.textContent = 0;
  currenScoreOne.textContent = 0;
  currenScoreTwo.textContent = 0;
  diceImage.classList.add("hidden");
}
btnNew.addEventListener("click", newGame);
btnRoll.addEventListener("click", handleRoll);
